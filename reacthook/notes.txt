useEffect is a functioncalled by react everytime the function updates or rerenders.

useEffect(()=>{
    console.log('Counter1 changed')
}, [counter]) Here useEffect will not be called everytime and will be called only if counter or whatever element that array contains is changed.

####useReducer with redux like dispatcher
import React, {useState, useReducer} from 'react';
import './App.css';

//when dispatch will be called
//state = {counter: 0, step: 1}
//step = {type: 'INCREMENT_COUNTER}
function randomFunction(state, action){
  switch(action.type){
    case 'INCREMENT_STEP':{
      return {
        ...state,
        step: state.step + action.payload
      }
    }
    case 'INCREMENT_COUNTER':{
      return {
        ...state,
        step: state.counter +  state.step
      }
    }
    default:{
      return state
    }
  }
}

function App() {
  // const [step, setStep] = useState(20)
  const [state, dispatch] = useReducer(randomFunction, {
    counter: 0,
    step: 1
  })

  function increaseIncrement(){
    dispatch({type: 'INCREMENT_STEP', payload: 1})
  }
  function decrementIncrement(){
    dispatch({type: 'INCREMENT_STEP', payload: -1})
  }
  function increaseCounter(){
    dispatch({type: 'INCREMENT_COUNTER'})
  }
  return (
    <div className="App">
        <h1 onClick={()=>increaseCounter}>Counter: {state.counter}</h1>
        <h2>Current step: {state.step}</h2>
        <button onClick={increaseIncrement}>Increase the step</button>
        <button onClick={decrementIncrement}>Decrease the step</button>

    </div>
  );
}

export default App;
##useMemo
usememo is used when it is computationally expensive. Unlike useState where if a object is passed everytime it renders it points to a different location in the Memory in useMemo it points to same location in the Memory until there is a change in the dependency array.

import React, {useMemo, useState, useCallback} from 'react';
import './App.css';


function App() {
  
  const [state, setState] = useState(0)

  const value = useMemo(()=>{
    console.log('I am inside useMemo')// Memoization and will be called only once
    return 100
  }, [])

  const obj2 = useCallback(()=>{
    console.log('Hello there')
    return <h1>Memoized component</h1>
  }, [state])
  
  const value2 = 100

  return(
    <div className='App'>
      <h1>Hello World - {value} | {value2}</h1>
      <h2 onClick={()=>setState(Math.random())}>State Variable - {state}</h2>
    </div>
  )
 
}

export default App;
